  person: {
    firstName: 'Ian',

    birthday: {

    }
  },
  verifications: {
    email: true
  },
  activity: {
    created: {
      timestamp: '2019-04-29T01:57:47Z',
      timestampUNIX: 1556503067000,
    },

      timestamp: '2019-04-29T01:57:47Z',
      timestampUNIX: 1556503067000,
    },
  },
  auth: {
    uid: '',
    email: '',
  },
  roles: {
    admin: false,
    developer: false,

      }

  },

    privateKey: 'api_1908-ea7ad440-bfa8-11e9-b4c3-f32ee92844a3',

    },
  },

    expires: {
      timestamp: '2111-01-01T00:00:00Z',
      timestampUNIX: 1556503067000,
    },    
  },
  // temp: false,
}
